常用组件动画 API

继承自BaseUI  Panel 事件
function Panel_Shop:On_Show( argTable)       --当打开时
    self.CG.alpha = 1
end
function Panel_Shop:On_Close()       --当关闭时
    self.CG.alpha = 0
end

CanvasGroup 相关
    self.CG.alpha = 1                     --透明度
    self.CG:DOFade(0, 0.6)                --透明度动画
    self.CG.interactable = true             --打开交互
    self.CG.blocksRaycasts = true           --打开射线拦截
备注：CG = CanvasGroup

RectTransform 相关
    self.uitable.ImagePro_PiPei.Rt.anchoredPosition = Vector2(-1314, 0)       --Rt位置
    self.uitable.ImagePro_PiPei.Rt:DOAnchorPos(Vector2(0, 0), 0.3)            --Rt动画

Text 相关
    self.togglePro.Text_lable.text = data.name                              --Text 内容
    self. togglePro.Text_lable.color = Color.black  /Color.white            --Text 颜色

DoTween 相关
--DoTween 的动画
    local tweener = self.uitable.ImagePro_PiPei.Rt:DOAnchorPos(Vector2(0, 0), 0.3)
--DoTween 的动画曲线
    tweener:SetEase(CS.DG.Tweening.Ease.OutBack)
--DoTween 的反向使用
    tweener:SetAutoKill(false)
    tweener:PlayBackwards()
--DoTween 的事件回调
    tweener:OnComplete(function()
        self.canvas_group.alpha = 0
    end)
